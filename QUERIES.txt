create table SALARIES(
ID NUMBER ,   
employee_id NUMBER  NOT NULL,
salary  NUMBER  NOT NULL,
PROJECT_ID NUMBER 
)


create table PROJECTS(
ID NUMBER,
Code NUMBER NOT NULL,
Budget NUMBER NOT NULL,
DATEDATE_OF_BEGINNING DATE,
DATE_OF_ENDIN DATE

)

create table EMPLOYEES(
ID NUMBER,
MANAGER_ID NUMBER ,
Date of joining NUMBER NOT  NULL,
full_name VARCHAR2 (255 BYTE)
)


SELECT * FROM EMPLOYEES
ORDER BY ID DESC

SELECT employee_id , salary    FROM SALARIES
ORDER BY employee_id 


SELECT * FROM EMPLOYEES
WHERE full_name LIKE '%O'
